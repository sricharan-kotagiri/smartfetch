CHANGE 2: Added shop detection hook
  useEffect(() => {
    const checkShop = async () => {
      // Get authenticated user
      const { data: { user } } = await supabase.auth.getUser()
      
      // Get shopkeeper using user_id
      const { data: shopkeeper } = await supabase
        .from('shopkeepers')
        .select('id')
        .eq('user_id', user.id)
        .maybeSingle()  // ✅ Key: maybeSingle() not single()
      
      // Get shop using shopkeeper.id
      const { data: shop } = await supabase
        .from('shops')
        .select('*')
        .eq('shopkeeper_id', shopkeeper.id)  // ✅ Key: shopkeeper.id not user.id
        .maybeSingle()
      
      // Redirect if shop exists
      if (shop) {
        navigate('/shopkeeper-dashboard')
      }
    }
    checkShop()
  }, [])

CHANGE 3: Fixed redirect after shop creation
  Before: setTimeout(() => navigate('/dashboard'), 1500)
  After:  navigate('/shopkeeper-dashboard')

================================================================================
                          WHY .maybeSingle()?
================================================================================

.single() (OLD - CAUSES 406 ERROR):
  - Throws error if 0 rows found
  - Throws error if 2+ rows found
  - Only works if exactly 1 row found
  - Result: 406 error in console

.maybeSingle() (NEW - CORRECT):
  - Returns null if 0 rows found (no error)
  - Returns row if 1 row found
  - Throws error if 2+ rows found
  - Result: No 406 error
